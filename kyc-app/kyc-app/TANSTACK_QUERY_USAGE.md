### Retry Logic

```javascript
const { data } = useQuery({
  queryKey: ['events'],
  queryFn: fetchEvents,
  retry: (failureCount, error) => {
    // Don't retry on 404
    if (error.status === 404) return false;
    
    // Retry up to 3 times
    return failureCount < 3;
  },
  retryDelay: (attemptIndex) => {
    // Exponential backoff
    return Math.min(1000 * 2 ** attemptIndex, 30000);
  },
});
```
